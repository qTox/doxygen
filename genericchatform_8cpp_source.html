518 bool GenericChatForm::needsToHideName(ChatLogIdx idx) const
519 {
520  // If the previous message is not rendered we should show the name
521  // regardless of other constraints
522  auto itemBefore = messages.find(idx - 1);
523  if (itemBefore == messages.end()) {
524  return false;
525  }
526 
527  const auto& prevItem = chatLog.at(idx - 1);
528  const auto& currentItem = chatLog.at(idx);
529 
530  // Always show the * in the name field for action messages
531  if (currentItem.getContentType() == ChatLogItem::ContentType::message
532  && currentItem.getContentAsMessage().message.isAction) {
533  return false;
534  }
535 
536  qint64 messagesTimeDiff = prevItem.getTimestamp().secsTo(currentItem.getTimestamp());
537  return currentItem.getSender() == prevItem.getSender()
538  && messagesTimeDiff < chatWidget->repNameAfter;
539 }

647 void GenericChatForm::loadHistoryTo(const QDateTime &time)
648 {

678 bool GenericChatForm::loadHistoryFrom(const QDateTime &time)
679 {
680  chatWidget->setScroll(false);
681  auto begin = chatLog.getFirstIdx();
682  if (time.isNull()) {
683  begin = messages.rbegin()->first;
684  } else {
685  begin = firstItemAfterDate(time.date(), chatLog);
686  }
687 
688  const auto end = chatLog.getNextIdx() < begin + DEF_NUM_MSG_TO_LOAD
689  ? chatLog.getNextIdx()
690  : begin + DEF_NUM_MSG_TO_LOAD;
691 
692  // The chatLog.getNextIdx() is usually 1 more than the idx on last "messages" item
693  // so if we have nothing to load, "add" is equal 1
694  if (end - begin <= 1) {
695  chatWidget->setScroll(true);
696  return false;
697  }
698 
699  renderMessages(begin, end);

991 void GenericChatForm::handleSearchResult(SearchResult result, SearchDirection direction)
992 {
993  if (!result.found) {
994  emit messageNotFoundShow(direction);
995  return;
996  }
997 
998  disableSearchText();
999 
1000  searchResult = result;
1001 
1002  auto searchIdx = result.pos.logIdx;
1003 
1004  auto firstRenderedIdx = messages.begin()->first;
1005  auto endRenderedIdx = messages.rbegin()->first;
1006 
1007  if (direction == SearchDirection::Up) {
1008  if (searchIdx < firstRenderedIdx) {
1009  if (searchIdx - chatLog.getFirstIdx() > DEF_NUM_MSG_TO_LOAD / 2) {
1010  firstRenderedIdx = searchIdx - DEF_NUM_MSG_TO_LOAD / 2;
1011  } else {

1013  }
1014  }
1015

1017  endRenderedIdx = firstRenderedIdx + DEF_NUM_MSG_TO_LOAD;
1018  }
1019  } else {
1020  if (searchIdx < firstRenderedIdx) {
1021  firstRenderedIdx = searchIdx;
1022  }
1023 
1024  if (firstRenderedIdx == searchIdx || searchIdx > endRenderedIdx) {
1025  if (searchIdx + DEF_NUM_MSG_TO_LOAD > chatLog.getNextIdx()) {
1026  endRenderedIdx = chatLog.getNextIdx();
1027  } else {
1028  endRenderedIdx = searchIdx + DEF_NUM_MSG_TO_LOAD;
1029  }
1030  }
1031 
1032  if (endRenderedIdx - firstRenderedIdx > DEF_NUM_MSG_TO_LOAD) {
1033  if (endRenderedIdx - chatLog.getFirstIdx() > DEF_NUM_MSG_TO_LOAD) {
1034  firstRenderedIdx = endRenderedIdx - DEF_NUM_MSG_TO_LOAD;
1035  } else {
1036  firstRenderedIdx = chatLog.getFirstIdx();
1037  }
1038  }
1039  }

1042  || endRenderedIdx > messages.rbegin()->first)) {
1043  chatWidget->clear();
1044  messages.clear();
1045

1047  endRenderedIdx = firstRenderedIdx;
