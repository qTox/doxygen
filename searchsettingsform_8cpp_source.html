57 ParameterSearch SearchSettingsForm::getParameterSearch()
58 {
59  ParameterSearch ps;
60 
61  if (ui->regularRadioButton->isChecked()) {
62  ps.filter = FilterSearch::Regular;
63  } else if (ui->registerCheckBox->isChecked() && ui->wordsOnlyRadioButton->isChecked()) {
64  ps.filter = FilterSearch::RegisterAndWordsOnly;
65  } else if (ui->registerCheckBox->isChecked() && ui->regularRadioButton->isChecked()) {
66  ps.filter = FilterSearch::RegisterAndRegular;
67  } else if (ui->registerCheckBox->isChecked()) {
68  ps.filter = FilterSearch::Register;
69  } else if (ui->wordsOnlyRadioButton->isChecked()) {
70  ps.filter = FilterSearch::WordsOnly;
71  } else {
72  ps.filter = FilterSearch::None;
73  }
74 
75  switch (ui->startSearchComboBox->currentIndex()) {
76  case 0:
77  ps.period = PeriodSearch::WithTheEnd;
78  break;
79  case 1:
80  ps.period = PeriodSearch::WithTheFirst;
81  break;
82  case 2:
83  ps.period = PeriodSearch::AfterDate;
84  break;
85  case 3:
86  ps.period = PeriodSearch::BeforeDate;
87  break;
88  default:
89  ps.period = PeriodSearch::WithTheEnd;
90  break;
91  }
92 
93  ps.date = startDate;
94  ps.isUpdate = isUpdate;
95  isUpdate = false;
96 
97  return ps;
98 }
