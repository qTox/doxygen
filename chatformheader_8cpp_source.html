242 void ChatFormHeader::updateCallButtons(bool online, bool audio, bool video)
243 {
244  const bool audioAvaliable = online && (mode & Mode::Audio);
245  const bool videoAvaliable = online && (mode & Mode::Video);
246  if (!audioAvaliable) {
247  callState = CallButtonState::Disabled;
248  } else if (video) {
249  callState = CallButtonState::Disabled;
250  } else if (audio) {
251  callState = CallButtonState::InCall;
252  } else {
253  callState = CallButtonState::Avaliable;
254  }
255 
256  if (!videoAvaliable) {
257  videoState = CallButtonState::Disabled;
258  } else if (video) {
259  videoState = CallButtonState::InCall;
260  } else if (audio) {
261  videoState = CallButtonState::Disabled;
262  } else {
263  videoState = CallButtonState::Avaliable;
264  }
265 
266  updateButtonsView();
267 }
