177 void ChatLog::layout(int start, int end, qreal width)
178 {
179  if (lines.empty()) {
180  return;
181  }
182 
183  qreal h = 0.0;
184 
185  // Line at start-1 is considered to have the correct position. All following lines are
186  // positioned in respect to this line.
187  if (start - 1 >= 0)

189 
190  start = clamp<int>(start, 0, lines.size());
191  end = clamp<int>(end + 1, 0, lines.size());
192 
193  for (int i = start; i < end; ++i) {
194  ChatLine* l = lines[i].get();
195 
196  l->layout(width, QPointF(0.0, h));
197  h += l->sceneBoundingRect().height() + lineSpacing;
