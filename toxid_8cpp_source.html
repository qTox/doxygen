243 bool ToxId::isValid() const
244 {
245  if (toxId.length() != TOX_ADDRESS_SIZE) {
246  return false;
247  }
248 
249  const int size = TOX_PUBLIC_KEY_SIZE + NOSPAM_BYTES;
250 
251  QByteArray data = toxId.left(size);
252  QByteArray checksum = toxId.right(CHECKSUM_BYTES);
253  QByteArray calculated(CHECKSUM_BYTES, 0x00);
254 
255  for (int i = 0; i < size; i++) {
256  calculated[i % 2] = calculated[i % 2] ^ data[i];
257  }
258 
259  return calculated == checksum;
260 }
