79 QString secondsToDHMS(quint32 duration)
80 {
81  QString res;
82  QString cD = ChatForm::tr("Call duration: ");
83  quint32 seconds = duration % 60;
84  duration /= 60;
85  quint32 minutes = duration % 60;
86  duration /= 60;
87  quint32 hours = duration % 24;
88  quint32 days = duration / 24;
89 
90  // I assume no one will ever have call longer than a month
91  if (days) {
92  return cD + res.asprintf("%dd%02dh %02dm %02ds", days, hours, minutes, seconds);
93  }
94 
95  if (hours) {
96  return cD + res.asprintf("%02dh %02dm %02ds", hours, minutes, seconds);
97  }
98 
99  if (minutes) {
100  return cD + res.asprintf("%02dm %02ds", minutes, seconds);
101  }
102 
103  return cD + res.asprintf("%02ds", seconds);
104 }

505 void ChatForm::dropEvent(QDropEvent* ev)
506 {
507  if (!ev->mimeData()->hasUrls()) {
508  return;
509  }
510 
511  for (const QUrl& url : ev->mimeData()->urls()) {
512  QFileInfo info(url.path());
513  QFile file(info.absoluteFilePath());
514 
515  QString urlString = url.toString();
516  if (url.isValid() && !url.isLocalFile()
517  && urlString.length() < static_cast<int>(tox_max_message_length())) {
518  messageDispatcher.sendMessage(false, urlString);
519 
520  continue;
521  }
522 
523  QString fileName = info.fileName();
524  if (!file.exists() || !file.open(QIODevice::ReadOnly)) {
525  info.setFile(url.toLocalFile());
526  file.setFileName(info.absoluteFilePath());
527  if (!file.exists() || !file.open(QIODevice::ReadOnly)) {
528  QMessageBox::warning(this, tr("Unable to open"),
529  tr("qTox wasn't able to open %1").arg(fileName));
530  continue;
531  }
532  }
533 
534  file.close();
535  if (file.isSequential()) {
536  QMessageBox::critical(nullptr, tr("Bad idea"),
537  tr("You're trying to send a sequential file, "
538  "which is not going to work!"));
539  continue;
540  }
541 
542  if (info.exists()) {
543  core.getCoreFile()->sendFile(f->getId(), fileName, info.absoluteFilePath(), info.size());
544  }
545  }

568 void ChatForm::sendImage(const QPixmap& pixmap)
569 {
570  QDir(Settings::getInstance().getPaths().getAppDataDirPath()).mkpath("images");
571 
572  // use ~ISO 8601 for screenshot timestamp, considering FS limitations
573  // https://en.wikipedia.org/wiki/ISO_8601
574  // Windows has to be supported, thus filename can't have `:` in it :/
575  // Format should be: `qTox_Screenshot_yyyy-MM-dd HH-mm-ss.zzz.png`
576  QString filepath = QString("%1images%2qTox_Image_%3.png")
577  .arg(Settings::getInstance().getPaths().getAppDataDirPath())
578  .arg(QDir::separator())
579  .arg(QDateTime::currentDateTime().toString("yyyy-MM-dd HH-mm-ss.zzz"));
580  QFile file(filepath);
581 
582  if (file.open(QFile::ReadWrite)) {
583  pixmap.save(&file, "PNG");
584  qint64 filesize = file.size();
585  file.close();
586  QFileInfo fi(file);
587  CoreFile* coreFile = core.getCoreFile();
588  coreFile->sendFile(f->getId(), fi.fileName(), fi.filePath(), filesize);
589  } else {
590  QMessageBox::warning(this,
591  tr("Failed to open temporary file", "Temporary file for screenshot"),
592  tr("qTox wasn't able to save the screenshot"));
593  }
594 }
