237 QString FileTransferWidget::getHumanReadableSize(qint64 size)
238 {
239  static const char* suffix[] = {"B", "KiB", "MiB", "GiB", "TiB"};
240  int exp = 0;
241 
242  if (size > 0) {
243  exp = std::min(static_cast<int>(log(size) / log(1024)), static_cast<int>(sizeof(suffix) / sizeof(suffix[0]) - 1));
244  }
245 
246  return QString().setNum(size / pow(1024, exp), 'f', exp > 1 ? 2 : 0).append(suffix[exp]);
247 }

320 void FileTransferWidget::updateFileProgress(ToxFile const& file)
321 {
322  switch (file.status) {
323  case ToxFile::INITIALIZING:
324  break;
325  case ToxFile::PAUSED:
326  fileProgress.resetSpeed();
327  break;
328  case ToxFile::TRANSMITTING: {
329  if (!fileProgress.needsUpdate()) {
330  break;
331  }
332 
333  fileProgress.addSample(file);
334  auto speed = fileProgress.getSpeed();
335  auto progress = fileProgress.getProgress();
336  auto remainingTime = fileProgress.getTimeLeftSeconds();
337 
338  ui->progressBar->setValue(static_cast<int>(progress * 100.0));
339 
340  // update UI
341  if (speed > 0) {
342  // ETA
343  QTime toGo = QTime(0, 0).addSecs(remainingTime);
344  QString format = toGo.hour() > 0 ? "hh:mm:ss" : "mm:ss";
345  ui->etaLabel->setText(toGo.toString(format));
346  } else {
347  ui->etaLabel->setText("");
348  }
349 
350  ui->progressLabel->setText(getHumanReadableSize(speed) + "/s");
351  break;
352  }
353  case ToxFile::BROKEN:
354  case ToxFile::CANCELED:
355  case ToxFile::FINISHED: {
356  ui->progressBar->hide();
357  ui->progressLabel->hide();
358  ui->etaLabel->hide();
359  break;
360  }
361  default:
362  qCritical() << "Invalid file status";
363  assert(false);
364  }

501 void FileTransferWidget::showPreview(const QString& filename)
502 {
503  static const QStringList previewExtensions = {"png", "jpeg", "jpg", "gif", "svg",
504  "PNG", "JPEG", "JPG", "GIF", "SVG"};
505 
506  if (previewExtensions.contains(QFileInfo(filename).suffix())) {
507  // Subtract to make border visible
508  const int size = qMax(ui->previewButton->width(), ui->previewButton->height()) - 4;
509 
510  QFile imageFile(filename);
511  if (!imageFile.open(QIODevice::ReadOnly)) {
512  return;
513  }
514 
515  const QByteArray imageFileData = imageFile.readAll();
516  QImage image = QImage::fromData(imageFileData);
517  auto orientation = ExifTransform::getOrientation(imageFileData);
518  image = ExifTransform::applyTransformation(image, orientation);
519 
520  const QPixmap iconPixmap = scaleCropIntoSquare(QPixmap::fromImage(image), size);
521 
522  ui->previewButton->setIcon(QIcon(iconPixmap));
523  ui->previewButton->setIconSize(iconPixmap.size());
524  ui->previewButton->show();
525  // Show mouseover preview, but make sure it's not larger than 50% of the screen
526  // width/height
527  const QRect desktopSize = QApplication::desktop()->geometry();
528  const int maxPreviewWidth{desktopSize.width() / 2};
529  const int maxPreviewHeight{desktopSize.height() / 2};
530  const QImage previewImage = [&image, maxPreviewWidth, maxPreviewHeight]() {
531  if (image.width() > maxPreviewWidth || image.height() > maxPreviewHeight) {
532  return image.scaled(maxPreviewWidth, maxPreviewHeight, Qt::KeepAspectRatio,
533  Qt::SmoothTransformation);
534  } else {
535  return image;
536  }
537  }();
538 
539  QByteArray imageData;
540  QBuffer buffer(&imageData);
541  buffer.open(QIODevice::WriteOnly);
542  previewImage.save(&buffer, "PNG");

563 QPixmap FileTransferWidget::scaleCropIntoSquare(const QPixmap& source, const int targetSize)
564 {
565  QPixmap result;
566 
567  // Make sure smaller-than-icon images (at least one dimension is smaller) will not be
568  // upscaled
569  if (source.width() < targetSize || source.height() < targetSize) {
570  result = source;
571  } else {
572  result = source.scaled(targetSize, targetSize, Qt::KeepAspectRatioByExpanding,
573  Qt::SmoothTransformation);
574  }
575 
576  // Then, image has to be cropped (if needed) so it will not overflow rectangle
577  // Only one dimension will be bigger after Qt::KeepAspectRatioByExpanding
578  if (result.width() > targetSize) {
579  return result.copy((result.width() - targetSize) / 2, 0, targetSize, targetSize);
580  } else if (result.height() > targetSize) {
581  return result.copy(0, (result.height() - targetSize) / 2, targetSize, targetSize);
582  }
583 
584  // Picture was rectangle in the first place, no cropping
585  return result;
586 }
