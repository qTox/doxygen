39 void Translator::translate(const QString& localeName)
40 {
41  QMutexLocker locker{&lock};
42 
43  if (!core_translator)
44  core_translator = new QTranslator();
45 
46  if (!app_translator)
47  app_translator = new QTranslator();
48 
49  // Remove old translations
50  QCoreApplication::removeTranslator(core_translator);
51  QApplication::removeTranslator(app_translator);
52 
53  // Load translations
54  QString locale = localeName.isEmpty() ? QLocale::system().name().section('_', 0, 0) : localeName;
55 
56  if (locale != "en") {
57  if (core_translator->load(locale, ":translations/")) {
58  qDebug() << "Loaded translation" << locale;
59 
60  // System menu translation
61  QString s_locale = "qt_" + locale;
62  QString location = QLibraryInfo::location(QLibraryInfo::TranslationsPath);
63  if (app_translator->load(s_locale, location)) {
64  QApplication::installTranslator(app_translator);
65  qDebug() << "System translation loaded" << locale;
66  } else {
67  qDebug() << "System translation not loaded" << locale;
68  }
69 
70  // Application translation
71  QCoreApplication::installTranslator(core_translator);
72  } else {
73  qDebug() << "Error loading translation" << locale;
74  }
75  }
