57 Time getTimeBucket(const QDateTime& date)
58 {
59  if (date == QDateTime()) {
60  return Time::Never;
61  }
62 
63  QDate today = QDate::currentDate();
64  // clang-format off
65  const QMap<Time, QDate> dates {
66  { Time::Today, today.addDays(0) },
67  { Time::Yesterday, today.addDays(-1) },
68  { Time::ThisWeek, today.addDays(-6) },
69  { Time::ThisMonth, today.addMonths(-1) },
70  { Time::Month1Ago, today.addMonths(-2) },
71  { Time::Month2Ago, today.addMonths(-3) },
72  { Time::Month3Ago, today.addMonths(-4) },
73  { Time::Month4Ago, today.addMonths(-5) },
74  { Time::Month5Ago, today.addMonths(-6) },
75  };
76  // clang-format on
77 
78  for (Time time : dates.keys()) {
79  if (dates[time] <= date.date()) {
80  return time;
81  }
82  }
83 
84  return Time::LongAgo;
85 }

450 void FriendListWidget::cycleContacts(GenericChatroomWidget* activeChatroomWidget, bool forward)
451 {
452  if (!activeChatroomWidget) {
453  return;
454  }
455 
456  int index = -1;
457  FriendWidget* friendWidget = qobject_cast<FriendWidget*>(activeChatroomWidget);
458 
459  if (mode == SortingMode::Activity) {
460  if (!friendWidget) {
461  return;
462  }
463 
464  const auto activityTime = getActiveTimeFriend(friendWidget->getFriend());
465  index = static_cast<int>(getTimeBucket(activityTime));
466  QWidget* widget = activityLayout->itemAt(index)->widget();
467  CategoryWidget* categoryWidget = qobject_cast<CategoryWidget*>(widget);
468 
469  if (categoryWidget == nullptr || categoryWidget->cycleContacts(friendWidget, forward)) {
470  return;
471  }
472 
473  index += forward ? 1 : -1;
474 
475  for (;;) {
476  // Bounds checking.
477  if (index < 0) {
478  index = LAST_TIME;
479  continue;
480  } else if (index > LAST_TIME) {
481  index = 0;
482  continue;
483  }
484 
485  auto* widget = activityLayout->itemAt(index)->widget();
486  categoryWidget = qobject_cast<CategoryWidget*>(widget);
487 
488  if (categoryWidget != nullptr) {
489  if (!categoryWidget->cycleContacts(forward)) {
490  // Skip empty or finished categories.
491  index += forward ? 1 : -1;
492  continue;
493  }
494  }
495 
496  break;
497  }
498 
499  return;
500  }
501 
502  QLayout* currentLayout = nullptr;
503  CircleWidget* circleWidget = nullptr;
504 
505  if (friendWidget != nullptr) {
506  const ToxPk& pk = friendWidget->getFriend()->getPublicKey();
507  uint32_t circleId = Settings::getInstance().getFriendCircleID(pk);
508  circleWidget = CircleWidget::getFromID(circleId);
509  if (circleWidget != nullptr) {
510  if (circleWidget->cycleContacts(friendWidget, forward)) {
511  return;
512  }
513 
514  index = circleLayout->indexOfSortedWidget(circleWidget);
515  currentLayout = circleLayout->getLayout();
516  } else {
517  currentLayout = listLayout->getLayoutOnline();
518  index = listLayout->indexOfFriendWidget(friendWidget, true);
519  if (index == -1) {
520  currentLayout = listLayout->getLayoutOffline();
521  index = listLayout->indexOfFriendWidget(friendWidget, false);
522  }
523  }
524  } else {
525  GroupWidget* groupWidget = qobject_cast<GroupWidget*>(activeChatroomWidget);
526  if (groupWidget != nullptr) {
527  currentLayout = groupLayout.getLayout();
528  index = groupLayout.indexOfSortedWidget(groupWidget);
529  } else {
530  return;
531  }
532  }
533 
534  index += forward ? 1 : -1;
535 
536  for (;;) {
537  // Bounds checking.
538  if (index < 0) {
539  currentLayout = nextLayout(currentLayout, forward);
540  index = currentLayout->count() - 1;
541  continue;
542  } else if (index >= currentLayout->count()) {
543  currentLayout = nextLayout(currentLayout, forward);
544  index = 0;
545  continue;
546  }
547 
548  // Go to the actual next index.
549  if (currentLayout == listLayout->getLayoutOnline()
550  || currentLayout == listLayout->getLayoutOffline()
551  || currentLayout == groupLayout.getLayout()) {
552  GenericChatroomWidget* chatWidget =
553  qobject_cast<GenericChatroomWidget*>(currentLayout->itemAt(index)->widget());
554 
555  if (chatWidget != nullptr)
556  emit chatWidget->chatroomWidgetClicked(chatWidget);
557 
558  return;
559  } else if (currentLayout == circleLayout->getLayout()) {
560  circleWidget = qobject_cast<CircleWidget*>(currentLayout->itemAt(index)->widget());
561  if (circleWidget != nullptr) {
562  if (!circleWidget->cycleContacts(forward)) {
563  // Skip empty or finished circles.
564  index += forward ? 1 : -1;
565  continue;
566  }
567  }
568  return;
569  } else {
570  return;
571  }
572  }
573 }
