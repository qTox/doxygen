299 bool Settings::verifyProxySettings(const QCommandLineParser& parser)
300 {
301  QString IPv6SettingString = parser.value("I").toLower();
302  QString LANSettingString = parser.value("L").toLower();
303  QString UDPSettingString = parser.value("U").toLower();
304  QString proxySettingString = parser.value("proxy").toLower();
305  QStringList proxySettingStrings = proxySettingString.split(":");
306 
307  const QString SOCKS5 = QStringLiteral("socks5");
308  const QString HTTP = QStringLiteral("http");
309  const QString NONE = QStringLiteral("none");
310  const QString ON = QStringLiteral("on");
311  const QString OFF = QStringLiteral("off");
312 
313  // Check for incompatible settings
314  bool activeProxyType = false;
315 
316  if (parser.isSet("P")) {
317  activeProxyType = proxySettingStrings[0] == SOCKS5 || proxySettingStrings[0] == HTTP;
318  }
319 
320  if (parser.isSet("I")) {
321  if (!(IPv6SettingString == ON || IPv6SettingString == OFF)) {
322  qCritical() << "Unable to parse IPv6 setting.";
323  return false;
324  }
325  }
326 
327  if (parser.isSet("U")) {
328  if (!(UDPSettingString == ON || UDPSettingString == OFF)) {
329  qCritical() << "Unable to parse UDP setting.";
330  return false;
331  }
332  }
333 
334  if (parser.isSet("L")) {
335  if (!(LANSettingString == ON || LANSettingString == OFF)) {
336  qCritical() << "Unable to parse LAN setting.";
337  return false;
338  }
339  }
340  if (activeProxyType && UDPSettingString == ON) {
341  qCritical() << "Cannot set UDP on with proxy.";
342  return false;
343  }
344 
345  if (activeProxyType && LANSettingString == ON) {
346  qCritical() << "Cannot use LAN discovery with proxy.";
347  return false;
348  }
349 
350  if (LANSettingString == ON && UDPSettingString == OFF) {
351  qCritical() << "Incompatible UDP/LAN settings.";
352  return false;
353  }
354 
355  if (parser.isSet("P")) {
356  if (proxySettingStrings[0] == NONE) {
357  // slightly lazy check here, accepting 'NONE[:.*]' is fine since no other
358  // arguments will be investigated when proxy settings are applied.
359  return true;
360  }
361  // Since the first argument isn't 'none', verify format of remaining arguments
362  if (proxySettingStrings.size() != 3) {
363  qCritical() << "Invalid number of proxy arguments.";
364  return false;
365  }
366 
367  if (!(proxySettingStrings[0] == SOCKS5 || proxySettingStrings[0] == HTTP)) {
368  qCritical() << "Unable to parse proxy type.";
369  return false;
370  }
371 
372  // TODO(Kriby): Sanity check IPv4/IPv6 addresses/hostnames?
373 
374  int portNumber = proxySettingStrings[2].toInt();
375  if (!(portNumber > 0 && portNumber < 65536)) {
376  qCritical() << "Invalid port number range.";
377  }
378  }
379  return true;
380 }

388 bool Settings::applyCommandLineOptions(const QCommandLineParser& parser)
389 {
390  if (!verifyProxySettings(parser)) {
391  return false;
392  }
393 
394  QString IPv6Setting = parser.value("I").toUpper();
395  QString LANSetting = parser.value("L").toUpper();
396  QString UDPSetting = parser.value("U").toUpper();
397  QString proxySettingString = parser.value("proxy").toUpper();
398  QStringList proxySettings = proxySettingString.split(":");
399 
400  const QString SOCKS5 = QStringLiteral("SOCKS5");
401  const QString HTTP = QStringLiteral("HTTP");
402  const QString NONE = QStringLiteral("NONE");
403  const QString ON = QStringLiteral("ON");
404  const QString OFF = QStringLiteral("OFF");
405 
406 
407  if (parser.isSet("I")) {
408  enableIPv6 = IPv6Setting == ON;
409  qDebug() << QString("Setting IPv6 %1.").arg(IPv6Setting);
410  }
411 
412  if (parser.isSet("P")) {
413  qDebug() << QString("Setting proxy type to %1.").arg(proxySettings[0]);
414 
415  quint16 portNumber = 0;
416  QString address = "";
417 
418  if (proxySettings[0] == NONE) {
419  proxyType = ICoreSettings::ProxyType::ptNone;
420  } else {
421  if (proxySettings[0] == SOCKS5) {
422  proxyType = ICoreSettings::ProxyType::ptSOCKS5;
423  } else if (proxySettings[0] == HTTP) {
424  proxyType = ICoreSettings::ProxyType::ptHTTP;
425  } else {
426  qCritical() << "Failed to set valid proxy type";
427  assert(false); // verifyProxySettings should've made this impossible
428  }
429 
430  forceTCP = true;
431  enableLanDiscovery = false;
432 
433  address = proxySettings[1];
434  portNumber = static_cast<quint16>(proxySettings[2].toInt());
435  }
436 
437 
438  proxyAddr = address;
439  qDebug() << QString("Setting proxy address to %1.").arg(address);
440  proxyPort = portNumber;
441  qDebug() << QString("Setting port number to %1.").arg(portNumber);
442  }
443 
444  if (parser.isSet("U")) {
445  bool shouldForceTCP = UDPSetting == OFF;
446  if (!shouldForceTCP && proxyType != ICoreSettings::ProxyType::ptNone) {
447  qDebug() << "Cannot use UDP with proxy; disable proxy explicitly with '-P none'.";
448  } else {
449  forceTCP = shouldForceTCP;
450  qDebug() << QString("Setting UDP %1.").arg(UDPSetting);
451  }
452 
453  // LANSetting == ON is caught by verifyProxySettings, the OFF check removes needless debug
454  if (shouldForceTCP && !(LANSetting == OFF) && enableLanDiscovery) {
455  qDebug() << "Cannot perform LAN discovery without UDP; disabling LAN discovery.";
456  enableLanDiscovery = false;
457  }
458  }
459 
460  if (parser.isSet("L")) {
461  bool shouldEnableLAN = LANSetting == ON;
462 
463  if (shouldEnableLAN && proxyType != ICoreSettings::ProxyType::ptNone) {
464  qDebug()
465  << "Cannot use LAN discovery with proxy; disable proxy explicitly with '-P none'.";
466  } else if (shouldEnableLAN && forceTCP) {
467  qDebug() << "Cannot use LAN discovery without UDP; enable UDP explicitly with '-U on'.";
468  } else {
469  enableLanDiscovery = shouldEnableLAN;
470  qDebug() << QString("Setting LAN Discovery %1.").arg(LANSetting);
471  }
472  }
473  return true;
474 }
