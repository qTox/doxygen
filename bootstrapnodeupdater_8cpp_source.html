53 void jsonNodeToDhtServer(const QJsonObject& node, QList<DhtServer>& outList)
54 {
55  // first check if the node in question has all needed fields
56  bool found = true;
57  for (const auto& key : NodeFields::neededFields) {
58  found &= node.contains(key);
59  }
60 
61  if (!found) {
62  qDebug() << "Node is missing required fields.";
63  return;
64  }
65 
66  // only use nodes that provide at least UDP connection
67  if (!node[NodeFields::status_udp].toBool(false)) {
68  return;
69  }
70 
71  const QString public_key = node[NodeFields::public_key].toString({});
72  const int port = node[NodeFields::port].toInt(-1);
73 
74  // nodes.tox.chat doesn't use empty strings for empty addresses
75  QString ipv6_address = node[NodeFields::ipv6].toString({});
76  if (ipv6_address == emptyAddress) {
77  ipv6_address = QString{};
78  }
79 
80  QString ipv4_address = node[NodeFields::ipv4].toString({});
81  if (ipv4_address == emptyAddress) {
82  ipv4_address = QString{};
83  }
84 
85  if (ipv4_address.isEmpty() && ipv6_address.isEmpty()) {
86  qWarning() << "Both ipv4 and ipv4 addresses are empty for" << public_key;
87  }
88 
89  const QString maintainer = node[NodeFields::maintainer].toString({});
90 
91  if (port < 1 || port > std::numeric_limits<uint16_t>::max()) {
92  qDebug() << "Invalid port in nodes list:" << port;
93  return;
94  }
95  const quint16 port_u16 = static_cast<quint16>(port);
96 
97  if (!public_key.contains(ToxPkRegEx)) {
98  qDebug() << "Invalid public key in nodes list" << public_key;
99  return;
100  }
101 
102  DhtServer server;
103  server.statusUdp = true;
104  server.statusTcp = node[NodeFields::status_udp].toBool(false);
105  server.userId = public_key;
106  server.port = port_u16;
107  server.maintainer = maintainer;
108  server.ipv4 = ipv4_address;
109  server.ipv6 = ipv6_address;
110  outList.append(server);
111  return;
112 }

114 QList<DhtServer> jsonToNodeList(const QJsonDocument& nodeList)
115 {
116  QList<DhtServer> result;
117 
118  if (!nodeList.isObject()) {
119  qWarning() << "Bootstrap JSON is missing root object";
120  return result;
121  }
122 
123  QJsonObject rootObj = nodeList.object();
124  if (!(rootObj.contains(jsonNodeArrayName) && rootObj[jsonNodeArrayName].isArray())) {
125  qWarning() << "Bootstrap JSON is missing nodes array";
126  return result;
127  }
128  QJsonArray nodes = rootObj[jsonNodeArrayName].toArray();
129  for (const QJsonValueRef node : nodes) {
130  if (node.isObject()) {
131  jsonNodeToDhtServer(node.toObject(), result);
132  }
133  }
134 
135  return result;
136 }

138 QList<DhtServer> loadNodesFile(QString file)
139 {
140  QFile nodesFile{file};
141  if (!nodesFile.open(QIODevice::ReadOnly | QIODevice::Text)) {
142  qWarning() << "Couldn't read bootstrap nodes";
143  return {};
144  }
145 
146  QString nodesJson = nodesFile.readAll();
147  nodesFile.close();
148 
149  auto jsonDoc = QJsonDocument::fromJson(nodesJson.toUtf8());
150  if (jsonDoc.isNull()) {
151  qWarning() << "Failed to parse JSON document";
152  return {};
153  }
154 
155  return jsonToNodeList(jsonDoc);
156 }

158 QByteArray serialize(QList<DhtServer> nodes)
159 {
160  QJsonArray jsonNodes;
161  for (auto& node : nodes) {
162  QJsonObject nodeJson;
163  nodeJson.insert(NodeFields::status_udp, node.statusUdp);
164  nodeJson.insert(NodeFields::status_tcp, node.statusTcp);
165  nodeJson.insert(NodeFields::ipv4, node.ipv4);
166  nodeJson.insert(NodeFields::ipv6, node.ipv6);
167  nodeJson.insert(NodeFields::public_key, node.userId);
168  nodeJson.insert(NodeFields::port, node.port);
169  nodeJson.insert(NodeFields::maintainer, node.maintainer);
170  jsonNodes.append(nodeJson);
171  }
172  QJsonObject rootObj;
173  rootObj.insert("nodes", jsonNodes);
174 
175  QJsonDocument doc{rootObj};
176  return doc.toJson(QJsonDocument::Indented);

190 QList<DhtServer> BootstrapNodeUpdater::getBootstrapnodes()
191 {
192  auto userFilePath = paths.getUserNodesFilePath();
193  if (!QFile(userFilePath).exists()) {
194  qInfo() << "Bootstrap node list not found, creating one with default nodes.";
195  // deserialize and reserialize instead of just copying to strip out any unnecessary json, making it easier for

200  QFile outFile(userFilePath);

202  outFile.write(serializedNodes.data(), serializedNodes.size());
203  outFile.close();
