44 void ChatTextEdit::keyPressEvent(QKeyEvent* event)
45 {
46  int key = event->key();
47  if ((key == Qt::Key_Enter || key == Qt::Key_Return) && !(event->modifiers() & Qt::ShiftModifier)) {
48  emit enterPressed();
49  return;
50  }
51  if (key == Qt::Key_Tab) {
52  if (event->modifiers())
53  event->ignore();
54  else {
55  emit tabPressed();
56  event->ignore();
57  }
58  return;
59  }
60  if (key == Qt::Key_Up && this->toPlainText().isEmpty()) {
61  this->setPlainText(lastMessage);
62  this->setFocus();
63  this->moveCursor(QTextCursor::MoveOperation::End, QTextCursor::MoveMode::MoveAnchor);
64  return;
65  }
66  if (event->matches(QKeySequence::Paste) && pasteIfImage(event)) {
67  return;
68  }
69  emit keyPressed();
70  QTextEdit::keyPressEvent(event);
71 }
