36 void ToxFileProgress::addSample(ToxFile const& file)
37 {
38  QTime now = QTime::currentTime();
39  qint64 dt = lastTick.msecsTo(now); // ms
40 
41  if (dt < 1000) {
42  return;
43  }
44 
45  // ETA, speed
46  qreal deltaSecs = dt / 1000.0;
47 
48  // (can't use ::abs or ::max on unsigned types substraction, they'd just overflow)
49  quint64 deltaBytes = file.bytesSent > lastBytesSent ? file.bytesSent - lastBytesSent
50  : lastBytesSent - file.bytesSent;
51  qreal bytesPerSec = static_cast<int>(static_cast<qreal>(deltaBytes) / deltaSecs);
52 
53  // Update member variables
54  meanIndex = meanIndex % TRANSFER_ROLLING_AVG_COUNT;
55  meanData[meanIndex++] = bytesPerSec;
56 
57  double meanBytesPerSec = 0.0;
58  for (size_t i = 0; i < TRANSFER_ROLLING_AVG_COUNT; ++i) {
59  meanBytesPerSec += meanData[i];
60  }
61  meanBytesPerSec /= static_cast<qreal>(TRANSFER_ROLLING_AVG_COUNT);
62 
63  lastTick = now;
64 
65  progress = static_cast<double>(file.bytesSent) / static_cast<double>(file.filesize);
66  speedBytesPerSecond = meanBytesPerSec;
67  timeLeftSeconds = (file.filesize - file.bytesSent) / getSpeed();
68 
69  lastBytesSent = file.bytesSent;
70 }
