69 std::unique_ptr<ToxOptions> ToxOptions::makeToxOptions(const QByteArray& savedata,
70  const ICoreSettings* s)
71 {
72  Tox_Options* tox_opts = tox_options_new(nullptr);
73 
74  if (!tox_opts) {
75  qWarning() << "Failed to create Tox_Options";
76  return {};
77  }
78 
79  // need to init proxyAddr here, because we need it to construct ToxOptions
80  const QString proxyAddr = s == nullptr ? QString{} : s->getProxyAddr();
81 
82  auto toxOptions = std::unique_ptr<ToxOptions>(new ToxOptions(tox_opts, proxyAddr.toUtf8()));
83  // register log first, to get messages as early as possible
84  tox_options_set_log_callback(*toxOptions, ToxLogger::onLogMessage);
85 
86  // savedata
87  tox_options_set_savedata_type(*toxOptions, savedata.isNull() ? TOX_SAVEDATA_TYPE_NONE
88  : TOX_SAVEDATA_TYPE_TOX_SAVE);
89  tox_options_set_savedata_data(*toxOptions, reinterpret_cast<const uint8_t*>(savedata.data()),
90  savedata.size());
91 
92  if(s == nullptr) {
93  qDebug() << "Using Tox default settings";
94  return toxOptions;
95  }
96 
97  // IPv6 needed for LAN discovery, but can crash some weird routers. On by default, can be
98  // disabled in options.
99  const bool enableIPv6 = s->getEnableIPv6();
100  bool forceTCP = s->getForceTCP();
101  // LAN requiring UDP is a toxcore limitation, ideally wouldn't be related
102  const bool enableLanDiscovery = s->getEnableLanDiscovery() && !forceTCP;
103  ICoreSettings::ProxyType proxyType = s->getProxyType();
104  quint16 proxyPort = s->getProxyPort();
105 
106  if (!enableLanDiscovery) {
107  qWarning() << "Core starting without LAN discovery. Peers can only be found through DHT.";
108  }
109  if (enableIPv6) {
110  qDebug() << "Core starting with IPv6 enabled";
111  } else if (enableLanDiscovery) {
112  qWarning() << "Core starting with IPv6 disabled. LAN discovery may not work properly.";
113  }
114 
115  // No proxy by default
116  tox_options_set_proxy_type(*toxOptions, TOX_PROXY_TYPE_NONE);
117  tox_options_set_proxy_host(*toxOptions, nullptr);
118  tox_options_set_proxy_port(*toxOptions, 0);
119 
120  if (proxyType != ICoreSettings::ProxyType::ptNone) {
121  if (static_cast<uint32_t>(proxyAddr.length()) > tox_max_hostname_length()) {
122  qWarning() << "Proxy address" << proxyAddr << "is too long";
123  } else if (!proxyAddr.isEmpty() && proxyPort > 0) {
124  qDebug() << "Using proxy" << proxyAddr << ":" << proxyPort;
125  // protection against changings in Tox_Proxy_Type enum
126  if (proxyType == ICoreSettings::ProxyType::ptSOCKS5) {
127  tox_options_set_proxy_type(*toxOptions, TOX_PROXY_TYPE_SOCKS5);
128  } else if (proxyType == ICoreSettings::ProxyType::ptHTTP) {
129  tox_options_set_proxy_type(*toxOptions, TOX_PROXY_TYPE_HTTP);
130  }
131 
132  tox_options_set_proxy_host(*toxOptions, toxOptions->getProxyAddrData());
133  tox_options_set_proxy_port(*toxOptions, proxyPort);
134 
135  if (!forceTCP) {
136  qDebug() << "Proxy and UDP enabled. This is a security risk. Forcing TCP only.";
137  forceTCP = true;
138  }
139  }
140  }
141 
142  // network options
143  tox_options_set_udp_enabled(*toxOptions, !forceTCP);
144  tox_options_set_ipv6_enabled(*toxOptions, enableIPv6);
145  tox_options_set_local_discovery_enabled(*toxOptions, enableLanDiscovery);
146  tox_options_set_start_port(*toxOptions, 0);
147  tox_options_set_end_port(*toxOptions, 0);
148 
149  return toxOptions;
150 }
