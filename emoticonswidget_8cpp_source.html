63  // create pages

67  pageLayout->addItem(new QSpacerItem(0, 0, QSizePolicy::Minimum, QSizePolicy::Expanding),
68  maxRows, 0);

103 
104  qobject_cast<QGridLayout*>(stack.widget(currPage)->layout())->addWidget(button, row, col);
105 
106  ++col;
107  ++currItem;
108 
109  // next row

114
